# useradd -d /home/username -m -p password -s /bin/bash -g group -G grops username

-d - домашний каталог
-m - создать домашний каталог
-p - пароль
-U - создать группу с таким же именем как и у пользователя
-s - /bin/sh - базовый шелл, будет выбран по-умолчанию
     /bin/bash - продвинутый шелл
	 /bin/false - запретить использовать шелл
-g - главная группа пользователя, должна существовать.
-G - группы в которые пользователь входит

$ cat /etc/passwd
список всех пользователей

$ cat /etc/group
список групп

# groups username
все группы пользователя в которые он входит, где первая - его главная группа

# userdel user
удалить пользователя

$ su - username (su -l username для ubuntu 14.04)
войти от имени другого пользователя, сменив свое окружение на его
если входит root, то пароль не запрашивается

# usermod -g group user
изменить первичную группу пользователя

# usermod -a -G group user
добавить пользователя в группу


НАСТРОЙКА SUDO ДЛЯ ДОБАВЛЕННОГО ПОЛЬЗОВАТЕЛЯ

# usermod -a -G sudo user
добавить пользователя в группу sudo

# visudo
открыть настройки sudo

%sudo   ALL=(ALL:ALL) ALL
заменить на
%sudo   ALL=(ALL:ALL) NOPASSWD:ALL
в результате у пользователя не будет запрошен пароль, что важно в ssh
